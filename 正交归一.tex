% 正交归一

\section{正交归一}
\begin{itemize}
	\item \(\braket{f}{f}=1\), \(f\) 是归一的
	\item \(\braket{f}{g}=0\)，\(f\)和\(g\) 正交
	\item \(\braket{f}{f}=1 \qq{and} \braket{g}{g}=1 \qq{and} \braket{f}{g}=0\)，\(f\)和\(g\) 是正交归一的。
\end{itemize}
